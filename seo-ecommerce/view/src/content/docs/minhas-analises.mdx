# Minhas Análises

Esta página lista as análises que você salvou estando autenticado (login simples localStorage). Nenhum dado vai para o servidor no modo local.

<div id="saved-analyses-root" style="margin-top:1rem;"></div>

<script>
(function(){
  const root = document.getElementById('saved-analyses-root');
  function render(){
    const user = localStorage.getItem('la-user');
    if(!user){
      root.innerHTML = `<p>Você não está logado. Use o formulário do Link Analyzer para entrar e salvar análises.</p>`;
      return;
    }
    const raw = localStorage.getItem('la-saved');
    const list = raw? JSON.parse(raw): [];
    if(!list.length){
      root.innerHTML = `<p>Nenhuma análise salva ainda.</p>`;return;
    }
    const html = [`<p><strong>Usuário:</strong> ${user} (<button id="la-logout" style="font-size:12px;">Logout</button>)</p>`,'<table style="width:100%;border-collapse:collapse;">','<thead><tr><th style="text-align:left;padding:4px 6px;border-bottom:1px solid #ddd;">URL</th><th style="text-align:left;padding:4px 6px;border-bottom:1px solid #ddd;">Data</th><th style="text-align:left;padding:4px 6px;border-bottom:1px solid #ddd;">Ações</th></tr></thead><tbody>'];
    list.forEach((item,i)=>{
      html.push(`<tr><td style="padding:4px 6px;">${item.url}</td><td style="padding:4px 6px;">${new Date(item.ts).toLocaleString()}</td><td style="padding:4px 6px;"><button data-i="${i}" class="la-view" style="font-size:11px;">Ver</button> <button data-i="${i}" class="la-del" style="font-size:11px;">Excluir</button></td></tr>`);
    });
    html.push('</tbody></table>');
    root.innerHTML = html.join('');
    root.querySelectorAll('.la-del').forEach(btn=>btn.addEventListener('click', e=>{
      const idx = +e.target.getAttribute('data-i');
      list.splice(idx,1); localStorage.setItem('la-saved', JSON.stringify(list)); render();
    }));
    root.querySelectorAll('.la-view').forEach(btn=>btn.addEventListener('click', e=>{
      const idx = +e.target.getAttribute('data-i');
      alert(JSON.stringify(list[idx].data, null, 2));
    }));
    const logout = document.getElementById('la-logout');
    logout.addEventListener('click',()=>{ localStorage.removeItem('la-user'); render(); });
  }
  render();
})();
</script>
